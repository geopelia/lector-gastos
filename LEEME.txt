# ejecutar docker contenedor solo
 docker run -it --rm prueba-php /bin/bash

# Compose (prender y apagar)
docker compose up -d
docker compose down

# conectarme a instancia
docker exec -i -t "php-prueba-app-1" /bin/bash

# rebuild imagen
 docker build -t prueba-php .
